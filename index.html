Python Script with Poem

import turtle

# Uma flor linda para uma pessoa linda

# Set up the turtle
t = turtle.Turtle()
t.speed(10)  # Set the drawing speed

# Function to draw a petal
def draw_petal():
    t.color("navy")  # Set the color to a darker shade of blue
    t.begin_fill()   # Begin filling the shape
    t.circle(50, 60) # Draw a part of a circle to form a petal
    t.left(120)      # Turn left
    t.circle(50, 60) # Draw another part of a circle to complete the petal
    t.end_fill()     # End filling the shape

# Function to draw a leaf
def draw_leaf():
    t.color("dark green")  # Set the color to dark green
    t.begin_fill()         # Begin filling the shape
    t.right(90)            # Turn right
    t.forward(50)          # Move forward
    t.left(120)            # Turn left
    t.forward(50)          # Move forward
    t.end_fill()           # End filling the shape

# Function to draw the rose
def draw_rose():
    for _ in range(6):
        draw_petal()    # Draw a petal
        t.left(60)      # Turn left to draw the next petal

    # Draw the center of the rose
    t.penup()
    t.goto(0, 0)
    t.pendown()
    t.color("yellow")  # Set the color to yellow for the center
    t.dot(20)          # Draw a smaller yellow dot as the center of the rose

    # Draw the leaves
    t.penup()
    t.goto(0, -10)     # Position the turtle for drawing leaves
    t.pendown()
    t.right(30)        # Turn the turtle right for better leaf orientation
    draw_leaf()        # Draw the first leaf
    t.left(120)        # Turn left
    draw_leaf()        # Draw the second leaf
    t.left(120)        # Turn left
    draw_leaf()        # Draw the third leaf

# Move the turtle to a good starting position
t.penup()
t.goto(0, -150)
t.pendown()

# Draw the rose
draw_rose()

# Insert the poem as a comment


# Hide the turtle
t.hideturtle()

# Keep the window open
turtle.done()
